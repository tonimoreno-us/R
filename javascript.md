# Breve sumario de instrucciones javascript interesantes

## join
const elements = ["1","2","3"];  
elements.join(); -> "1,2,3"  
elements.join(""); -> "123"  
elements.join("-"); -> "1-2-3"  
  
## toString
elements.toString(); -> "1,2,3"
